#include<math.h>

  double a,b,c,delta,x1,x2;
  scanf("%lf,%lf,%lf",&a,&b,&c);
  if(b*b-4*a*c<=0){
    printf("error\n");
  }

**定义二维数组**

例题 将一个二维数组a=
| 1 | 2 | 3 |
|---|---|---|
| 4 | 5 | 6 |

的行列元素互换，存到另一个二维数组b种并输出。
```
#include<stdio.h>
int main(void){
  int a[2][3]={{1,2,3},{4,5,6}};
  int b[3][2],i,j;
  for(i=0;i<=1;i++){
    for(j=0;j<=2;j++){
      b[j][i]=a[i][j];
    }
  }
  printf("数组b为:\n");
  for(j=0;j<=2;j++){
    for(i=0;i<=1;i++){
      printf("%d\t",b[j][i]);
    }
    printf("\n");
  }
  return 0;
}
```
